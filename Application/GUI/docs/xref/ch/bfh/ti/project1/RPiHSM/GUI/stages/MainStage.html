13      private Button addKeyButton, createKeySetButton, deleteKeySetButton, decryptButton, encryptButton, promoteDemoteButton, pubKeyButton, revokeButton, signButton, verifyButton;
14  
15      /**
16       * Generates all the graphical elements for the supported operations.
17       *
18       * @param serialHelper an instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI}
19       * @param userPath     user home directory on the Raspberry Pi
20       */
21      public MainStage(SerialHelperI serialHelper, String userPath) {
22          super(serialHelper, userPath);
23  
24          //creates the scene objects
25          this.sceneTitle = new Label(Constants.MAIN_STAGE_TITLE);
26          addKeyButton = new Button(Constants.MAIN_STAGE_BUTTON_ADDKEY);
27          createKeySetButton = new Button(Constants.MAIN_STAGE_BUTTON_CREATEKEYSET);
28          deleteKeySetButton = new Button(Constants.MAIN_STAGE_BUTTON_DELETEKEYSET);
29          decryptButton = new Button(Constants.MAIN_STAGE_BUTTON_DECRYPT);
30          encryptButton = new Button(Constants.MAIN_STAGE_BUTTON_ENCRYPT);
31          promoteDemoteButton = new Button(Constants.MAIN_STAGE_BUTTON_PROMOTE_DEMOTE);
32          pubKeyButton = new Button(Constants.MAIN_STAGE_BUTTON_PUBLICKEY);
33          revokeButton = new Button(Constants.MAIN_STAGE_BUTTON_REVOKE);
34          signButton = new Button(Constants.MAIN_STAGE_BUTTON_SIGN);
35          verifyButton = new Button(Constants.MAIN_STAGE_BUTTON_VERIFY);
36  
37          //Sets the buttons width
38          addKeyButton.setMinWidth(Constants.MAIN_MENU_BUTTON_SIZE);
39          createKeySetButton.setMinWidth(Constants.MAIN_MENU_BUTTON_SIZE);
40          deleteKeySetButton.setMinWidth(Constants.MAIN_MENU_BUTTON_SIZE);
41          decryptButton.setMinWidth(Constants.MAIN_MENU_BUTTON_SIZE);
42          encryptButton.setMinWidth(Constants.MAIN_MENU_BUTTON_SIZE);
43          promoteDemoteButton.setMinWidth(Constants.MAIN_MENU_BUTTON_SIZE);
44          pubKeyButton.setMinWidth(Constants.MAIN_MENU_BUTTON_SIZE);
45          revokeButton.setMinWidth(Constants.MAIN_MENU_BUTTON_SIZE);
46          signButton.setMinWidth(Constants.MAIN_MENU_BUTTON_SIZE);
47          verifyButton.setMinWidth(Constants.MAIN_MENU_BUTTON_SIZE);
48  
49          //Sets the actions for each button
50          //each button opens a new stage
51          addKeyButton.setOnMouseClicked(e -> {
52              this.hide();
53              new AddKeyCommandStage(serialHelper, userPath);
54          });
55          createKeySetButton.setOnMouseClicked(e -> {
56              this.hide();
57              new CreateKeySetCommandStage(serialHelper, userPath);
58          });
59          deleteKeySetButton.setOnMouseClicked(e -> {
60              this.hide();
61              new DeleteKeySetCommandStage(serialHelper, userPath);
62          });
63          encryptButton.setOnMouseClicked(e -> {
64              this.hide();
65              new EncryptDecryptCommandStage(serialHelper, userPath, true);
66          });
67          decryptButton.setOnMouseClicked(e -> {
68              this.hide();
69              new EncryptDecryptCommandStage(serialHelper, userPath, false);
70          });
71          promoteDemoteButton.setOnMouseClicked(e -> {
72              this.hide();
73              new PromoteDemoteCommandStage(serialHelper, userPath);
74          });
75          pubKeyButton.setOnMouseClicked(e -> {
76              this.hide();
77              new PublicKeyCommandStage(serialHelper, userPath);
78          });
79          revokeButton.setOnMouseClicked(e -> {
80              this.hide();
81              new RevokeCommandStage(serialHelper, userPath);
82          });
83          signButton.setOnMouseClicked(e -> {
84              this.hide();
85              new SignCommandStage(serialHelper, userPath);
86          });
87          verifyButton.setOnMouseClicked(e -> {
88              this.hide();
89              new VerifyCommandStage(serialHelper, userPath);
90          });
91  
92          //places the objects in the Grid

94          grid.add(addKeyButton, 0, 1);
95          grid.add(createKeySetButton, 1, 1);
96          grid.add(deleteKeySetButton, 2, 1);
97          grid.add(decryptButton, 0, 2);
98          grid.add(encryptButton, 1, 2);
99          grid.add(promoteDemoteButton, 2, 2);
100         grid.add(pubKeyButton, 0, 3);
101         grid.add(revokeButton, 1, 3);
102         grid.add(signButton, 2, 3);
103         grid.add(verifyButton, 1, 4);
104         grid.setHgap(30);
105         grid.setVgap(30);
106     }
