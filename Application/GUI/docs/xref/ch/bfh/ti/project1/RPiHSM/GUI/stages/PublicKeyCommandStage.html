17  /**
18   * <h1>PublicKeyCommandStage</h1>
19   * An implementation of {@link AbstractStage}. This stage allows the user to generate the public key by using the {@link PublicKey} class.
20   */
21  public class PublicKeyCommandStage extends AbstractStage {
22      private Button executeButton, directoryChooserButton;
23      private Label keySetLabel;
24      private TextField keySetTextField;
25      private Label results;
26      private DirectoryChooser destinationDirectoryChooser;
27      private File destinationFolder;
28  
29  
30      /**
31       * Generates all the graphical elements for the public key generation command.
32       *
33       * @param serialHelper an instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI}
34       * @param userPath     user home directory on the Raspberry Pi
35       */
36      public PublicKeyCommandStage(SerialHelperI serialHelper, String userPath) {
37          super(serialHelper, userPath);
38  
39          //creates the scene objects
40          this.sceneTitle = new Label(Constants.PUBLIC_KEY_COMMAND_TITLE);
41          executeButton = new Button(Constants.BUTTON_TEXT_PUBLICKEY);
42          directoryChooserButton = new Button(Constants.DIRECTORY_CHOOSER_SELECT_DESTINATION);
43          destinationDirectoryChooser = new DirectoryChooser();
44          destinationFolder = new File("");
45          keySetLabel = new Label(Constants.KEY_SET);
46          keySetTextField = new TextField();
47          results = new Label();
48  
49          //disables the file chooser if the key set does not exist
50          directoryChooserButton.disableProperty().bind(messages.textProperty().isEqualTo(Constants.KEYSET_EXISTS).not());
51          //disables the execute button if no file has been selected
52          executeButton.disableProperty().bind(messages.textProperty().isEqualTo(Constants.FILE_CHOOSER_DESTINATION_SELECTED).not());
53  
54          //On mouse click, deletes all the files of the destination folder and tries to get all public keys of the given key set
55          executeButton.setOnMouseClicked(e -> {
56              //delete all files of the destination folder
57              File[] keysToDelete = destinationFolder.listFiles();
58              for (int i = 0; i < keysToDelete.length; i++) {
59                  keysToDelete[i].delete();
60              }
61  
62              PublicKey pk = new PublicKey(serialHelper, userPath, keySetTextField.getText(), destinationFolder.getAbsolutePath());
63  
64              try {
65                  if (pk.generate()) {
66                  	clearElements();
67                      success(Constants.PUBLIC_KEY_SUCCESS);
68                      File[] keys = destinationFolder.listFiles();
69                      for (int i = 0; i < keys.length; i++) {//reads all files in the destination folder to the results Label
70                          results.setText(results.getText() + "\n\n" + keys[i].getAbsolutePath());
71                          results.setText(results.getText() + "\n" + Files.readAllLines(keys[i].toPath()));
72                      }
73                  } else {
74                      error(Constants.PUBLIC_KEY_NOT_SUCCESS);
75                  }
76              } catch (OperationNotSupportedException | KeySetNotAsymmetricException | KeySetIsEmptyException | IOException e1) {
77                  error(Constants.PUBLIC_KEY_NOT_SUCCESS);
78              }
79          });
80  
81          //On mouse clicked, sets the destination directory as the directory selected with the DirectoryChooser
82          directoryChooserButton.setOnMouseClicked(e -> {
83              File temp = destinationDirectoryChooser.showDialog(this);
84              if (temp != null) {
85                  destinationFolder = temp;
86                  success(Constants.FILE_CHOOSER_DESTINATION_SELECTED);
87              } else {
88                  error(Constants.FILE_CHOOSER_DESTINATION_NOT_SELECTED);
89              }
90          });
91  
92          //sets the width as maximal
93          results.minWidth(this.getWidth());
94          results.maxWidth(this.getWidth());
95  
96          //on focus out checks if the key set exists
97          keySetTextField.focusedProperty().addListener((obs, oldVal, newVal) -> {
98              if (!newVal) {
99                  this.checkKeyExistence(keySetTextField.getText());
100             }
101         });
102 
103         //places the objects in the Grid
104         addHeader(true);
105         grid.setGridLinesVisible(false);
106         grid.add(keySetLabel, 0, 0);
107         grid.add(keySetTextField, 1, 0);
108         grid.add(directoryChooserButton, 2, 0);
109         grid.add(executeButton, 3, 0);
110         grid.add(results, 0, 1, 4, 1);
111     }
112     
113     /**
114      * Method to reset the elements to its default values
115      */
116     private void clearElements() {
117         keySetTextField.clear();
118     }
119 }
