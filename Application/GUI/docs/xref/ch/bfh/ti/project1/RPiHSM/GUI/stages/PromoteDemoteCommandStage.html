12  import javax.naming.OperationNotSupportedException;

14  /**
15   * <h1>PromoteDemoteCommandStage</h1>
16   * An implementation of {@link AbstractStage}. This stage allows the user to promote or demote a key by using the {@link KeyStatus} class.
17   */
18  public class PromoteDemoteCommandStage extends AbstractStage {
19      private Button executeButton;
20      private Label keySetLabel, versionLabel, actionLabel;
21      private TextField keySetTextField, versionTextField;
22      private ComboBox<String> actionComboBox;
23  
24      /**
25       * Generates all the graphical elements for the promote/demote key command.
26       *
27       * @param serialHelper an instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI}
28       * @param userPath     user home directory on the Raspberry Pi
29       */
30      public PromoteDemoteCommandStage(SerialHelperI serialHelper, String userPath) {
31          super(serialHelper, userPath);
32  
33          //creates the scene objects
34          this.sceneTitle = new Label(Constants.PROMOTE_DEMOTE_COMMAND_TITLE);
35          executeButton = new Button(Constants.PROMOTE);
36          keySetLabel = new Label(Constants.KEY_SET);
37          keySetTextField = new TextField();
38          versionLabel = new Label(Constants.VERSION_LABEL);
39          versionTextField = new TextField();
40          actionComboBox = new ComboBox<>();
41          actionLabel = new Label(Constants.ACTION_LABEL);
42  
43          actionComboBox.getItems().addAll(Constants.PROMOTE, Constants.DEMOTE);
44          actionComboBox.getSelectionModel().selectFirst();
45  
46          executeButton.setDisable(true); //disabled at the beginning
47          //On mouse clicked checks if the action to perform is promote or demote, then tries to perform the action and displays the success or error messages
48          executeButton.setOnMouseClicked(e -> {
49              KeyStatus ks = new KeyStatus(serialHelper, userPath, keySetTextField.getText(), Integer.parseInt(versionTextField.getText()));
50              if (actionComboBox.getValue().equals(Constants.PROMOTE)) { //Action: promote
51                  try {
52                      if (ks.promote()){
53                      	clearElements();
54                      	success(Constants.PROMOTE_SUCCESS);
55                      }
56                      else{
57                      	error(Constants.PROMOTE_NOT_SUCCESS);
58                      }
59                  } catch (OperationNotSupportedException e1) {
60                      error(Constants.PROMOTE_NOT_SUCCESS);
61                  }
62              } else { //Action: demote
63                  try {
64                      if (ks.demote()){
65                      	clearElements();
66                      	success(Constants.DEMOTE_SUCCESS);
67                      }
68                      else{
69                      	error(Constants.DEMOTE_NOT_SUCCESS);
70                      }
71                  } catch (OperationNotSupportedException e1) {
72                      error(Constants.DEMOTE_NOT_SUCCESS);
73                  }
74              }
75          });
76  
77          //disables the execute button if the key set does not exist of if the key version is not a number
78          versionTextField.textProperty().addListener((obs, oldValue, newValue) -> {
79              executeButton.disableProperty().bind(Bindings
80                      .when((new SimpleBooleanProperty(
81                              versionTextField.getText().matches(Constants.NUMBER_FIELD_PATTERN)))
82                              .and(keySetTextField.textProperty().length().greaterThan(2)))
83                      .then(false)
84                      .otherwise(true).or(messages.textProperty().isEqualTo(Constants.KEYSET_EXISTS).not()));
85          });
86  
87          //on value changed, changes the text of execute button (promote or demote key)
88          actionComboBox.setOnAction(e -> executeButton.setText(actionComboBox.getSelectionModel().getSelectedItem() + " key"));
89  
90          //disables version textfield and action combo box if the key set does not exist
91          versionTextField.disableProperty().bind(messages.textProperty().isEqualTo(Constants.KEYSET_EXISTS).not());
92          actionComboBox.disableProperty().bind(messages.textProperty().isEqualTo(Constants.KEYSET_EXISTS).not());
93  
94          //on focus out checks if the key set exists
95          keySetTextField.focusedProperty().addListener((obs, oldVal, newVal) -> {
96              if (!newVal) {
97                  this.checkKeyExistence(keySetTextField.getText());
98              }
99          });
100 
101         //places the objects in the Grid
102         addHeader(true);
103         grid.setGridLinesVisible(false);
104         grid.add(actionLabel, 0, 0);
105         grid.add(actionComboBox, 1, 0);
106         grid.add(keySetLabel, 0, 1);
107         grid.add(keySetTextField, 1, 1);
108         grid.add(versionLabel, 0, 2);
109         grid.add(versionTextField, 1, 2);
110         grid.add(executeButton, 1, 4);
111     }
112     
113     /**
114      * Method to reset the elements to its default values
115      */
116     private void clearElements() {
117         keySetTextField.clear();
118         versionTextField.clear();
119         actionComboBox.getSelectionModel().select(Constants.PROMOTE);
120         executeButton.setText(Constants.PROMOTE + " key");
121     }
122 }
