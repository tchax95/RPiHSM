3   import ch.bfh.ti.project1.RPiHSM.IoT.Utils.Constants;
4   import org.keyczar.KeyczarTool;
5   
6   import java.io.File;
7   
8   /**
9    * <h1>CreateKey</h1>
10   * Creates a new key by using the <a href="https://github.com/google/keyczar/wiki/KeyczarTool">KeyczarTool</a>.
11   */
12  public class CreateKey implements CommandI {
13  
14      private SerialHelperI serialHelper;
15      private String keyPath;
16      private String status;
17      private int size;
18  
19      /**
20       * Parses the received parameters.
21       *
22       * @param commands     parameters for the key creation ([0] userPath, [1] name, [2] status, [3] size)
23       * @param serialHelper an instance of {@link SerialHelperI}
24       */
25      public CreateKey(String[] commands, SerialHelperI serialHelper) {
26          keyPath = commands[0] + Constants.DIRECTORY + commands[1];
27          status = commands[2];
28          size = Integer.parseInt(commands[3]);
29          this.serialHelper = serialHelper;
30  
31      }
32  
33  
34  
35      /**
36       * Creates a new key using the <a href="https://github.com/google/keyczar/wiki/KeyczarTool">KeyczarTool</a>.
37       * If the key cannot be created an Exception is thrown and it cannot be catch.
38       * To check if the command has been successful, the number of files in the key set directory are checked.
39       * If the creation is successful the {@link SerialHelperI#ready()} is called otherwise {@link SerialHelperI#error()}.
40       * Prerequisites: the given key set must exist (to check it use {@link KeyExistence}).
41       */
42      @Override
43      public void execute() {
44  
45          //Check if key set already exists
46  
47          File varTmpDir = new File(keyPath + "//meta");
48  
49          long count = varTmpDir.getParentFile().list().length; // number of file before the key creation
50  
51          KeyczarTool.main(new String[]{"addkey", "--location=" + keyPath, "--status=" + status, (size != 0) ? "--size=" + size : ""}); // if there are errors an exception is thrown and cannot be catch (external application)
52  
53          if (varTmpDir.getParentFile().list().length <= count) {
54              serialHelper.error(); //key set creation failed due to key size
55          } else {
56              serialHelper.ready();
57          }
58  
59      }
60  
61  }
