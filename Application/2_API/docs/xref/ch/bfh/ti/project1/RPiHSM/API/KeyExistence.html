3   import javax.naming.OperationNotSupportedException;

5   /**
6    * <h1>KeyExistence</h1>
7    * Ask the IoT application using the {@link SerialHelperI} to check fi a key set with the given name exists.
8    */
9   public final class KeyExistence {
10  
11      private SerialHelperI serialHelper;
12      private String userPath;
13      private String name;
14  
15      /**
16       * Sets {@link SerialHelperI} instance to communicate with the IoT application and the key existence parameters.
17       *
18       * @param serialHelper an instance of {@link SerialHelperI}
19       * @param userPath     home path of the IoT application user
20       * @param name         name of the key set
21       */
22      public KeyExistence(SerialHelperI serialHelper, String userPath, String name) {
23          this.serialHelper = serialHelper;
24          this.name = name;
25          this.userPath = userPath;
26      }
27  
28      /**
29       * Ask to the IoT application if the key set with the {@link CreateKey#name} exists.
30       *
31       * @return true if the key set exists, false if the {@link KeyExistence#name} is empty or the key set does not exist
32       * @throws OperationNotSupportedException if an error occurs in the {@link SerialHelperI}
33       */
34      public boolean keyexist() throws OperationNotSupportedException {
35          if(name.isEmpty()) return false;
36          serialHelper.writeLine(String.join(" ", new String[]{Constants.CHECK, userPath, name}));
37  
38          if (!serialHelper.status()) {
39              return false;
40          }
41          return true;
42      }
43  
44  }
