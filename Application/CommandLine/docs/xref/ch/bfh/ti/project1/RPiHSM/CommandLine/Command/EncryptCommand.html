8   import java.io.FileNotFoundException;
9   import java.util.ArrayList;
10  import java.util.List;
11  
12  import static ch.bfh.ti.project1.RPiHSM.CommandLine.Utils.Constants.*;
13  
14  
15  /**
16   * <h1>EncryptCommand</h1>
17   * Parses the command line parameters and use the {@link EncryptDecrypt} to encrypt a file on the Raspberry Pi.
18   */
19  public class EncryptCommand implements CommandI {
20  
21      private SerialHelper serialHelper;
22  
23      private String userPath;
24  
25      @Parameter(description = PARAMETERS_LIST)//JCommander requirements (all others parameters)
26      private List<String> command = new ArrayList<>();
27  
28      @Parameter(names = {NAME_COMPLETE_VALUE, NAME_SHORT_VALUE}, description = NAME_VALUE, required = true)
29      private String keySetName;
30  
31      @Parameter(names = {FILE_COMPLETE_VALUE, FILE_SHORT_VALUE}, description = FILE_ENCRYPT_VALUE, required = true)
32      private String filePath;
33  
34      /**
35       * Saves the userPath and  an instance of the {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI} for future use.
36       *
37       * @param userPath     user home directory on the Raspberry Pi
38       * @param serialHelper an instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI}
39       */
40      public EncryptCommand(String userPath, SerialHelper serialHelper) {
41          this.userPath = userPath;
42          this.serialHelper = serialHelper;
43      }
44  
45      /**
46       * Encrypts a file with the received parameters on the Raspberry Pi using {@link EncryptDecrypt}.
47       *
48       * @return a message that describes the result of the operation (error or success)
49       */
50      @Override
51      public String execute() {
52          EncryptDecrypt ed = new EncryptDecrypt(serialHelper, userPath, keySetName, filePath);
53          try {
54              if (ed.encrypt()) {
55                  return ENCRYPT_SUCCESS;
56              } else {
57                  return ENCRYPT_ERROR;
58              }
59          } catch (OperationNotSupportedException e) {
60              return UNSUPPORTED_OPERATION;
61          } catch (FileNotFoundException e) {
62              return FILE_NOT_FOUND;
63          }
64      }
65  
66      /**
67       * Returns a message that contains the right command syntax.
68       *
69       * @return right command syntax
70       */
71      @Override
72      public String print() {
73          return ENCRYPT_COMMAND;
74      }
75  }
