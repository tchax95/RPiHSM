18  /**
19   * <h1>CommandLine</h1>
20   * Command line application that use the RPiHSM-API.
21   * To parse the command line parameters the <a href="http://jcommander.org">JCommander</a> framework is used.
22   *
23   * @author Noli Manzoni, Sandro Tiago Carlao
24   * @version alpha 0.3
25   * @since 29.03.2017
26   */
27  public class CommandLine {
28      /**
29       * An instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI} is created. If errors occurs, the application is terminated.
30       * The application ask for the user authentication and use the {@link Login} to authenticate the credentials.
31       * If the user is authenticated a new {@link CommandI} instance is created thanks to the {@link CommandFactory}.
32       * Then the JCommander framework parses the command parameters and if there are no errors the {@link CommandI} is executed.
33       *
34       * @param args the command parameters (to more information look at {@link CommandI#print()})
35       */
36      public static void main(String[] args) {
37  
38  
39          SerialHelper serialHelper;
40          try {
41              serialHelper = new SerialHelper();
42  
43  
44              //asks user info
45              Console console = System.console();
46              String userName = console.readLine(USER_NAME);
47              char password[] = console.readPassword(PASSWORD);
48  
49              Login login = new Login(serialHelper, userName, new String(password));//performs login
50  
51              if (login.checkCredentials()) { //if user is authenticated
52  
53                  CommandFactory cmFactory = new CommandFactory();//factory
54                  CommandI cm = cmFactory.getCommand(args[0].trim().toLowerCase(), login.getUserPath(), serialHelper);
55  
56                  //Create new parsing for command line arguments
57                  try {
58                      new JCommander(cm, args);
59  
60  
61                      String message = cm.execute();//Execute command
62  
63  
64                      System.out.println(message);//print the returned message
65  
66                  } catch (ParameterException e) { // if the JCommand find some syntax errors
67                      System.out.println(ILLEGAL_ARGUMENT);
68                      System.out.println(cm.print());//print right syntax
69                  } finally {
70                      try {
71                          serialHelper.closeConnection(); //The serial connection is closed
72                      } catch (SerialPortException e) {
73                          System.out.println(SERIAL_PORT_ERROR);
74                      }
75                  }
76              } else {
77                  System.out.println(WRONG_CREDENTIALS);
78              }
79  
80          } catch (PortInUseException e) {
81              System.out.println(SERIAL_PORT_IN_USE);
82          } catch (UnsupportedCommOperationException e) {
83              System.out.println(UNSUPPORTED_COM_OPERATION);
84          } catch (SerialPortException e) {
85              System.out.println(SERIAL_PORT_ERROR);
86          } catch (OperationNotSupportedException e) {
87              System.out.println(UNSUPPORTED_OPERATION);
88          } finally {
89              System.exit(0); // System exist with status 0 -> all the errors are catch and printed to the user
90          }
91  
92      }
93  }
