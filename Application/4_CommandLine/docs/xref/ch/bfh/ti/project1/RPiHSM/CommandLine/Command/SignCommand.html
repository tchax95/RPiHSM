13  import java.util.List;

15  import static ch.bfh.ti.project1.RPiHSM.CommandLine.Utils.Constants.*;
16  
17  /**
18   * <h1>SignCommand</h1>
19   * Parses the command line parameters and use the {@link Sign} to create a signature on the Raspberry Pi.
20   */
21  public class SignCommand implements CommandI {
22  
23  
24      private SerialHelper serialHelper;
25  
26      private String userPath;
27  
28      @Parameter(description = PARAMETERS_LIST)//JCommander requirements (all others parameters)
29      private List<String> command = new ArrayList<>();
30  
31      @Parameter(names = {NAME_COMPLETE_VALUE, NAME_SHORT_VALUE}, description = NAME_VALUE, required = true)
32      private String keySetName;
33  
34      @Parameter(names = {FILE_COMPLETE_VALUE, FILE_SHORT_VALUE}, description = FILE_SIGN_VALUE, required = true)
35      private String filePath;
36  
37      /**
38       * Saves the userPath and  an instance of the {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI} for future use.
39       *
40       * @param userPath     user home directory on the Raspberry Pi
41       * @param serialHelper an instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI}
42       */
43      public SignCommand(String userPath, SerialHelper serialHelper) {
44          this.userPath = userPath;
45          this.serialHelper = serialHelper;
46      }
47  
48      /**
49       * Creates a signature of a given file with the given key set on the Raspberry Pi using {@link Sign}.
50       *
51       * @return a message that describes the result of the operation (error or success)
52       */
53      @Override
54      public String execute() {
55          int index = filePath.lastIndexOf('.');
56          File signatureFile = new File(filePath.substring(0, index) + "-signature.bin");
57          try {
58              FileUtils.copyFile(new File(filePath), signatureFile); //file copy
59          } catch (IOException e2) {
60              return FILE_COPY_ERROR;
61          }
62  
63          Sign s = new Sign(serialHelper, userPath, keySetName, signatureFile.getAbsolutePath());
64  
65          try {
66              if (s.sign()) {
67                  return SIGN_SUCCESS;
68              } else {
69                  return SIGN_ERROR;
70              }
71          } catch (OperationNotSupportedException e) {
72              return UNSUPPORTED_OPERATION;
73          } catch (FileNotFoundException e) {
74              return FILE_NOT_FOUND;
75          }
76  
77      }
78  
79      /**
80       * Returns a message that contains the right command syntax.
81       *
82       * @return right command syntax
83       */
84      @Override
85      public String print() {
86          return SIGN_COMMAND;
87      }
88  }
