5   import ch.bfh.ti.project1.RPiHSM.API.SerialHelper;

7   /**
8    * <h1>CommandFactory</h1>
9    * Creates a {@link CommandI} instance.
10   * This factory follows the Command Factory design pattern.
11   */
12  public class CommandFactory {
13  
14      /**
15       * Creates a {@link CommandI} instance by parsing the input value.
16       *
17       * @param command      command to execute (the {@link CommandI} instance)
18       * @param userPath     authenticated user home directory.
19       * @param serialHelper an instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI}
20       * @return a {@link CommandI} instance
21       */
22      public CommandI getCommand(String command, String userPath, SerialHelper serialHelper) {
23          if (command.equals(Constants.SET)) {
24              return new CreateKeySetCommand(userPath, serialHelper);
25          } else if (command.equals(Constants.KEY)) {
26              return new CreateKeyCommand(userPath, serialHelper);
27          } else if (command.equals(Constants.ENCRYPT)) {
28              return new EncryptCommand(userPath, serialHelper);
29          } else if (command.equals(Constants.DECRYPT)) {
30              return new DecryptCommand(userPath, serialHelper);
31          } else if (command.equals(Constants.SIGN)) {
32              return new SignCommand(userPath, serialHelper);
33          } else if (command.equals(Constants.VERIFY)) {
34              return new VerifyCommand(userPath, serialHelper);
35          } else if (command.equals(Constants.CHANGE)) {
36              return new KeyStatusCommand(userPath, serialHelper);
37          } else if (command.equals(Constants.PUB)) {
38              return new PublicKeyCommand(userPath, serialHelper);
39          } else if (command.equals(Constants.DELETE)) {
40              return new DeleteKeySetCommand(userPath, serialHelper);
41          } else if (command.equals(Constants.HELP)) {
42              return new HelpCommand();
43          } else {
44              return new HelpCommand();
45          }
46      }
47  }
