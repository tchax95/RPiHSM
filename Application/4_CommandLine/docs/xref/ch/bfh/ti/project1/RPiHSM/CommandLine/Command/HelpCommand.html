12   * Prints all the possible command for this command line application.
13   */
14  public class HelpCommand implements CommandI {
15  
16      @Parameter(description = PARAMETERS_LIST)//JCommander requirements (all others parameters)
17      private List<String> command = new ArrayList<>();
18  
19      /**
20       * Creates a message with all possible command line application commands.
21       *
22       * @return all possible command line application commands
23       */
24      @Override
25      public String execute() {
26          String newLine = System.getProperty("line.separator");
27          return "The following commands are supported:" + newLine +
28                  KEY_COMMAND + newLine +
29                  KEY_SET_COMMAND + newLine +
30                  DECRYPT_COMMAND + newLine +
31                  DELETE_KEY_SET_COMMAND + newLine +
32                  ENCRYPT_COMMAND + newLine +
33                  KEY_STATUS_COMMAND + newLine +
34                  PUBLIC_KEY_COMMAND + newLine +
35                  SIGN_COMMAND + newLine +
36                  VERIFY_COMMAND;
37      }
38  
39      /**
40       * Do nothing - this method cannot be called because no error can occur
41       *
42       * @return empty string
43       */
44      @Override
45      public String print() {
46          return "";
47      }//no errors can occur to call this method
48  }
