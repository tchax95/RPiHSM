25  /**
26   * <h1>AbstractStage</h1>
27   * The base stage of the application that have all the common elements.
28   */
29  public abstract class AbstractStage extends Stage {
30  
31      private VBox pane;
32  
33      protected Label messages;
34      protected GridPane grid;
35      protected Label sceneTitle;
36      protected String userPath;
37      protected SerialHelperI serialHelper;
38  
39      /**
40       * Generates all the common elements of the stages.
41       *
42       * @param serialHelper an instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI}
43       * @param userPath     user home directory on the Raspberry Pi
44       */
45      public AbstractStage(SerialHelperI serialHelper, String userPath) {
46          super();
47  
48          this.userPath = userPath;
49          this.serialHelper = serialHelper;
50  
51          this.setTitle(Constants.TITLE); //Application title
52  
53          //Creates the Label for error and success messages
54          messages = new Label();
55          messages.setTextAlignment(TextAlignment.CENTER);
56          messages.setPadding(new Insets(10));
57          messages.prefWidth(Constants.WIDTH);
58  
59          //Creates the GridPane to place the objects on the stage
60          grid = new GridPane();
61          grid.setAlignment(Pos.CENTER);
62          grid.setHgap(Constants.PADDING_SMALL);
63          grid.setVgap(Constants.PADDING_SMALL);
64          grid.setPadding(new Insets(Constants.PADDING_MEDIUM));
65  
66          //Creates the VBox
67          pane = new VBox();
68          pane.setAlignment(Pos.TOP_CENTER);
69          pane.getChildren().add(this.messages);
70          pane.getChildren().add(grid);
71          pane.setStyle(Constants.PANE_BACKGROUND);
72  
73          //sets and shows the scene
74          this.setScene(new Scene(pane, Constants.WIDTH, Constants.HEIGHT));
75          this.show();
76  
77          this.setOnCloseRequest(e -> {
78              if (serialHelper != null) {
79                  try {
80                      serialHelper.closeConnection(); //closes the serial connection while closing the application
81                  } catch (SerialPortException e1) {
82                      error(Constants.UNSUPPORTED_COM_OPERATION);
83                  }
84              }
85          });
86      }
87  
88  
89      /**
90       * Generates the header elements of the application.
91       *
92       * @param backToMenu true if a button to go back in the menu must be displayed otherwise false
93       */
94      protected void addHeader(boolean backToMenu) {
95          sceneTitle.setFont(Font.font(Constants.FONT, FontWeight.NORMAL, Constants.TITLE_FONT_SIZE));
96          sceneTitle.setTextFill(Paint.valueOf(Constants.HEADER_TITLE_COLOR));
97          ImageView logo = new ImageView(this.getClass().getResource(Constants.HEADER_IMAGE_PATH).toExternalForm());
98  
99          //Creates the GridPane for the header structure
100         GridPane header = new GridPane();
101         header.setMinSize(Constants.WIDTH, Constants.HEADER_HEIGHT);
102         header.setPadding(new Insets(15, 12, 15, 12));
103         header.setStyle(Constants.HEADER_BACKGROUND);
104         ColumnConstraints col1 = new ColumnConstraints();
105         ColumnConstraints col2 = new ColumnConstraints();
106         ColumnConstraints col3 = new ColumnConstraints();
107         col1.setPercentWidth(25);
108         col2.setPercentWidth(50);
109         col3.setPercentWidth(23);
110 
111         header.add(logo, 0, 0);
112         header.add(sceneTitle, 1, 0);
113         header.getColumnConstraints().addAll(col1, col2, col3);
114 
115         GridPane.setHalignment(logo, HPos.LEFT);
116         GridPane.setHalignment(sceneTitle, HPos.CENTER);
117 
118         if (backToMenu) { //Creates the backToMenu button if needed
119             Button menuButton = new Button(Constants.BACK_BUTTON_TEXT);
120             menuButton.setPrefSize(100, 20);
121             menuButton.setOnMouseClicked(e -> {
122                 this.hide();
123                 new MainStage(serialHelper, userPath);
124             });
125             header.add(menuButton, 2, 0);
126             GridPane.setHalignment(menuButton, HPos.RIGHT);
127         } else {
128             header.add(new Label(""), 2, 0);
129         }
130 
131         pane.getChildren().add(0, header); //adds the header on first position (on top)
132     }
133 
134     /**
135      * Shows an error in the {@link AbstractStage#messages} element.
136      *
137      * @param message to display
138      */
139     protected void error(String message) {
140         messages.setTextFill(Color.FIREBRICK);
141         messages.setText(message); //prints the error message in Firebrick (red)
142     }
143 
144     /**
145      * Shows a success message in the {@link AbstractStage#messages} element.
146      *
147      * @param message to display
148      */
149     protected void success(String message) {
150         messages.setTextFill(Color.GREEN);
151         messages.setText(message); //prints the success message in green
152     }
153 
154     /**
155      * Checks if a given key set exist using the {@link KeyExistence} class.
156      *
157      * @param keySet name to check
158      */
159     protected void checkKeyExistence(String keySet) {
160         KeyExistence ke = new KeyExistence(serialHelper, userPath, keySet);
161         try {
162             if (ke.keyexist()) success(Constants.KEYSET_EXISTS); //checks if the key set exists
163             else error(Constants.KEYSET_NOT_EXISTS);
164         } catch (OperationNotSupportedException e) {
165             error(Constants.KEYSET_NOT_EXISTS);
166         }
167     }
