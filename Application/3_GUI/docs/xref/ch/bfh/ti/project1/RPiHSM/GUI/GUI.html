13   * <h1>GUI</h1>

20  public class GUI extends Application {
21      /**
22       * Start point of the GUI application.
23       * An instance of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI} is created. If errors occurs, the error message is displayed.
24       * If the creation of {@link ch.bfh.ti.project1.RPiHSM.API.SerialHelperI} is successful the {@link LoginStage} is opened.
25       *
26       * @param primaryStage the primary stage of the application
27       */
28      @Override
29      public void start(Stage primaryStage) {
30          SerialHelper serialHelper;
31  
32          try {
33              serialHelper = new SerialHelper();
34              new LoginStage(serialHelper); //creates the first stage
35          } catch (PortInUseException | UnsupportedCommOperationException e) {
36              new ErrorStage(Constants.PORT_IN_USE); //if the port is in use
37          } catch (Exception e) {
38              new ErrorStage(Constants.PORT_NOT_CONNECTED); //if the port is not connected
39          }
40      }
41  
42      /**
43       * Used to start the GUI by calling the {@link Application#launch(String...)}.
44       *
45       * @param args no arguments are need for this application
46       */
47      public static void main(String[] args) {
48          launch(args);
49      }
50  }
