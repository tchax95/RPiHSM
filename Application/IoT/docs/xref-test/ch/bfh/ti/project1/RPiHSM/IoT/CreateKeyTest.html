16  /**
17   * Test all {@link CreateKey} functionality.
18   */
19  public class CreateKeyTest {
20      private static MockSerialHelper mockSerialHelper;
21      private static String path;
22      private static String name;
23  
24      @BeforeClass
25      /**
26       * Creates a folder to store the key sets.
27       */
28      public static void setUp() {
29          mockSerialHelper = new MockSerialHelper();
30          path = System.getProperty("user.dir");
31          name = "JUNIT_TEST_CREATE_KEY";
32          File file = new File(path + Constants.DIRECTORY);
33          file.mkdir();
34      }
35  
36      @AfterClass
37      /**
38       * Deletes the folder for the key sets.
39       */
40      public static void tearDown() throws IOException {
41          File file = new File(path + Constants.DIRECTORY);
42          FileUtils.deleteDirectory(file);
43      }
44  
45      @Test
46      /**
47       * Tests positive key creation.
48       */
49      public void testPositiveKeyCreation() {
50          CreateKey ck = new CreateKey(new String[]{path, name, "primary", "0"}, mockSerialHelper);
51          ck.execute();
52          assertTrue(mockSerialHelper.isReady());
53          assertFalse(mockSerialHelper.isError());
54      }
55  
56      @Test
57      /**
58       * Tests negative key creation (wrong size).
59       */
60      public void testNegativeKeyCreation() {
61          CreateKey ck = new CreateKey(new String[]{path, name, "primary", "1"}, mockSerialHelper); // a keyczar exception is thrown -> size cannot be 1
62          ck.execute();
63          assertFalse(mockSerialHelper.isReady());
64          assertTrue(mockSerialHelper.isError());
65      }
66  
67  
68      @Before
69      /**
70       * Creates a key set for the tests.
71       */
72      public void set() {
73          CreateKeySet cks = new CreateKeySet(new String[]{path, name, "crypt", ""}, mockSerialHelper);
74          cks.execute();
75          mockSerialHelper.reset();
76      }
77  
78      @After
79      /**
80       * Deletes the key set used for the tests.
81       */
82      public void tear() throws IOException {
83          mockSerialHelper.reset();
84          File file = new File(path + Constants.DIRECTORY + name);
85          FileUtils.deleteDirectory(file);
86      }
